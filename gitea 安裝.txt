gitea 安裝

# 取得最新的 image
docker pull gitea/gitea:latest

# 查看當前image list
docker images

# 建立gitea容器
docker volume create gitea-data
docker volume create gitea-config


這邊碰到問題是啟動的 port一直不能使用最後先使用以下才可以啟動

docker run -d --name=gitea -p 38888:3000 -p 2222:22 -v gitea-data:/data -v gitea-config:/etc/gitea --restart=always gitea/gitea:latest


# 也可以使用yml啟動 將以下內容放到 docker-compose.yml 檔案需要放在 gitea資料夾中
# 執行 docker compose up -d 可以運行
# 執行 docker compose down 關掉


version: "3"

services:
  gitea:
    image: gitea/gitea:latest
    container_name: gitea
    restart: always
    ports:
      - "3000:3000"   # Web UI
      - "2222:22"     # Git SSH 連線用
    volumes:
      - gitea-data:/data
      - gitea-config:/etc/gitea

volumes:
  gitea-data:
  gitea-config:

安裝後 http://localhost:38888/ 可以開啟，並自行新增一個 Repository 

# 嘗試建一個檔案 push to git
mkdir hello-gitea
cd hello-gitea
git init
echo "# Hello Gitea" > README.md
git add .
git commit -m "first commit"

commit 正常會出錯，因為沒有帳號信箱所以使用
git config --global user.name "neil"
git config --global user.email "lichung0205@gmail.com"

嘗試 push
git remote add origin http://localhost:38888/neil/hello-gitea.git
git push -u origin master
